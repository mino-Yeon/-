이벤트란?

특정 행위에 의해 발생하는 것입니다. 이벤트는 특정 조건을 만족하면 항상 발생하도록 설계되어 있습니다.

click input submit change keydown keyup focus blur mouseover scroll DOMContentLoaded

이벤트 핸들링

이벤트가 발생했을 때 처리하는 방법은 세 가지입니다.

1. 인라인 이벤트 핸들링

HTML 태그에 직접 이벤트를 작성하는 방식입니다.

<input id="inp" type="text" oninput="print()" />
<script>
  const input = document.getElementById("inp");
  function print() {
    console.log(input.value);
  }
</script>

2. 이벤트 핸들러 프로퍼티 방식

<input id="inp" type="text" />
<script>
  const input = document.getElementById("inp");
  input.oninput = function () {
    console.log(input.value);
  };
</script>
하나의 이벤트에 핸들러를 하나밖에 등록할 수 없습니다. 나중에 등록한 핸들러가 이전 것을 덮어씁니다. 실무에서는 거의 사용하지 않습니다.

3. addEventListener() — 권장

가장 많이 사용하는 방식입니다. 하나의 이벤트에 여러 핸들러를 등록할 수 있습니다.

<input id="inp" type="text" />
<script>
  const input = document.getElementById("inp");
  input.addEventListener("input", function () {
    console.log("input 이벤트 발생");
  });
</script>
React 등 프레임워크에서는 JSX 기반 인라인 방식(onClick, onChange)을 주로 사용합니다. 바닐라 JS에서는 addEventListener가 표준입니다.

이벤트 객체

event

이벤트가 발생하면 핸들러 함수에 이벤트 객체가 자동으로 전달됩니다.

주요 프로퍼티

element.addEventListener("click", function (event) {
  console.log(event.type);          // 이벤트 종류 ("click")
  console.log(event.target);        // 실제 이벤트가 발생한 요소
  console.log(event.currentTarget); // 핸들러가 등록된 요소
});

target vs currentTarget

event.target은 실제로 클릭된 요소, event.currentTarget은 핸들러가 등록된 요소입니다. 아래에서 각 버튼을 클릭해 차이를 확인해보세요.

// 부모(outer)에만 핸들러 등록
outer.addEventListener("click", function (e) {
  console.log(e.target);        // 실제로 클릭한 요소 (버튼 A or B or outer)
  console.log(e.currentTarget); // 항상 outer — 핸들러가 등록된 곳
});
🔍 데모 — target vs currentTarget

핸들러는 파란 영역(부모)에만 등록되어 있습니다. 각 버튼을 클릭해보세요.

📦 outer (핸들러 등록된 곳 — currentTarget)
▶ 이벤트 로그

preventDefault()

preventDefault

이벤트의 기본 동작을 막습니다. submit 이벤트의 기본 동작은 페이지 새로고침인데, preventDefault()로 막고 직접 처리할 수 있습니다.

기본 동작 그대로 — 폼 제출 시 페이지가 새로고침됩니다.

form.addEventListener("submit", function (e) {
  // preventDefault() 없음 → 페이지 새로고침 발생
});
🔴 데모 — 적용 전 (제출 시 새로고침 발생)

preventDefault() 적용 — 새로고침 없이 입력값을 직접 처리합니다.

form.addEventListener("submit", function (e) {
  e.preventDefault(); // 기본 동작(새로고침) 차단
  console.log("입력값:", input.value);
});
🟢 데모 — 적용 후 (새로고침 없음)
▶ 처리 결과

이벤트 버블링

stopPropagation

자식 요소에서 발생한 이벤트가 부모 요소로 전파되는 현상입니다. 아래 예시는 button → td → tr 3단계 구조에서 각 요소에 핸들러가 등록되어 있습니다.

<tr id="tr">
  <td id="td">
    <button id="btn">삭제</button>
  </td>
</tr>
tr.addEventListener("click",  () => console.log("tr 클릭"));
td.addEventListener("click",  () => console.log("td 클릭"));
btn.addEventListener("click", () => console.log("버튼 클릭"));
▶ 버튼 클릭 시 출력 순서
버튼 클릭
td 클릭    ← 버블링
tr 클릭    ← 버블링

버튼 클릭 시 이벤트가 td → tr 순으로 버블링되어 모든 핸들러가 실행됩니다.

const tr  = document.getElementById("tr");
const td  = document.getElementById("td");
const btn = document.getElementById("btn");

tr.addEventListener("click",  () => console.log("tr — 상세화면 이동"));
td.addEventListener("click",  () => console.log("td — 핸들러 실행"));
btn.addEventListener("click", () => console.log("버튼 — 삭제 처리"));
🔴 데모 — 버블링 발생 (버튼 클릭 시 tr까지 전파)
tr (최상위 부모) — 클릭 시 상세화면 이동
td (자식) 🗑 삭제
▶ 이벤트 로그

버튼 핸들러에 stopPropagation()을 추가하면 td · tr로의 전파가 차단됩니다.

const tr  = document.getElementById("tr");
const td  = document.getElementById("td");
const btn = document.getElementById("btn");

tr.addEventListener("click",  () => console.log("tr — 상세화면 이동"));
td.addEventListener("click",  () => console.log("td — 핸들러 실행"));
btn.addEventListener("click", (e) => {
  e.stopPropagation(); // td · tr로의 버블링 차단
  console.log("버튼 — 삭제 처리");
});
🟢 데모 — 버블링 차단 (버튼 핸들러만 실행)
tr (최상위 부모) — 클릭 시 상세화면 이동
td (자식) 🗑 삭제
▶ 이벤트 로그

자주 쓰는 이벤트

click
요소를 클릭했을 때 발생
btn.addEventListener("click", (e) => {
  console.log("클릭!", e.target.textContent);
});
—
submit
폼이 제출될 때. 항상 preventDefault()와 함께 사용
form.addEventListener("submit", (e) => {
  e.preventDefault();
  console.log(input.value);
});
—
input
입력값이 바뀔 때마다 실시간으로 발생
input.addEventListener("input", (e) => {
  console.log(e.target.value);
});
—
change
값이 변경되고 포커스를 잃었을 때 발생
select.addEventListener("change", (e) => {
  console.log(e.target.value);
});
—
keydown
키보드를 눌렀을 때. 어떤 키인지 확인 가능
input.addEventListener("keydown", (e) => {
  console.log(e.key);
});
—
focus · blur
포커스를 얻을 때 / 잃을 때
input.addEventListener("focus", () => { ... });
input.addEventListener("blur",  () => { ... });
—
mouseover · mouseout
마우스가 요소 위에 올라올 때 / 벗어날 때
el.addEventListener("mouseover", () => { ... });
el.addEventListener("mouseout",  () => { ... });
마우스를 올려보세요
—
DOMContentLoaded
HTML 파싱 완료 시. 스크립트를 <head>에 넣을 때 사용
document.addEventListener(
  "DOMContentLoaded", () => {
    console.log("DOM 준비 완료");
  }
);
—
